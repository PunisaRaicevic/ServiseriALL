-- WARNING: This schema is for context only and is not meant to be run.
-- Table order and constraints may not be valid for execution.

CREATE TABLE public.appliances (
  appliance_id uuid NOT NULL DEFAULT gen_random_uuid(),
  client_id uuid,
  appliance_name text NOT NULL,
  appliance_maker text,
  appliance_age integer,
  appliance_serial_number text,
  appliance_picture text,
  last_service_date date,
  next_service_date date,
  appliance_location text,
  appliance_install_date date,
  created_at timestamp with time zone DEFAULT now(),
  CONSTRAINT appliances_pkey PRIMARY KEY (appliance_id),
  CONSTRAINT appliances_client_id_fkey FOREIGN KEY (client_id) REFERENCES public.clients(client_id)
);
CREATE TABLE public.clients (
  client_id uuid NOT NULL DEFAULT gen_random_uuid(),
  client_name text NOT NULL,
  client_email text,
  client_phone text,
  client_address text,
  client_pib text,
  client_pdv text,
  client_account text,
  created_at timestamp with time zone DEFAULT now(),
  CONSTRAINT clients_pkey PRIMARY KEY (client_id)
);
CREATE TABLE public.documents (
  document_id uuid NOT NULL DEFAULT gen_random_uuid(),
  document_name text NOT NULL,
  document_url text,
  document_type text,
  uploaded_by uuid,
  created_at timestamp with time zone DEFAULT now(),
  CONSTRAINT documents_pkey PRIMARY KEY (document_id),
  CONSTRAINT documents_uploaded_by_fkey FOREIGN KEY (uploaded_by) REFERENCES auth.users(id)
);
CREATE TABLE public.profiles (
  id uuid NOT NULL,
  user_name text,
  created_at timestamp with time zone DEFAULT now(),
  CONSTRAINT profiles_pkey PRIMARY KEY (id),
  CONSTRAINT profiles_id_fkey FOREIGN KEY (id) REFERENCES auth.users(id)
);
CREATE TABLE public.reports (
  report_id uuid NOT NULL DEFAULT gen_random_uuid(),
  task_id uuid,
  user_id uuid,
  description text,
  photos ARRAY,
  spare_parts_used jsonb,
  work_duration integer,
  created_at timestamp with time zone DEFAULT now(),
  CONSTRAINT reports_pkey PRIMARY KEY (report_id),
  CONSTRAINT reports_task_id_fkey FOREIGN KEY (task_id) REFERENCES public.tasks(task_id),
  CONSTRAINT reports_user_id_fkey FOREIGN KEY (user_id) REFERENCES auth.users(id)
);
CREATE TABLE public.spare_parts (
  spare_part_id uuid NOT NULL DEFAULT gen_random_uuid(),
  spare_part_name text NOT NULL,
  spare_part_maker text,
  spare_part_detail text,
  spare_part_picture text,
  spare_part_quantity integer DEFAULT 0,
  created_at timestamp with time zone DEFAULT now(),
  CONSTRAINT spare_parts_pkey PRIMARY KEY (spare_part_id)
);
CREATE TABLE public.tasks (
  task_id uuid NOT NULL DEFAULT gen_random_uuid(),
  client_id uuid,
  appliance_id uuid,
  user_id uuid,
  status text DEFAULT 'pending'::text CHECK (status = ANY (ARRAY['pending'::text, 'in_progress'::text, 'completed'::text])),
  task_type text DEFAULT 'one-time'::text CHECK (task_type = ANY (ARRAY['one-time'::text, 'recurring'::text])),
  task_description text,
  task_due_date date,
  created_at timestamp with time zone DEFAULT now(),
  completed_at timestamp with time zone,
  report_id uuid,
  CONSTRAINT tasks_pkey PRIMARY KEY (task_id),
  CONSTRAINT tasks_client_id_fkey FOREIGN KEY (client_id) REFERENCES public.clients(client_id),
  CONSTRAINT tasks_appliance_id_fkey FOREIGN KEY (appliance_id) REFERENCES public.appliances(appliance_id),
  CONSTRAINT tasks_user_id_fkey FOREIGN KEY (user_id) REFERENCES auth.users(id)
);
